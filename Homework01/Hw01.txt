//The code below is written in Kotlin

fun sumOfEvenSquares(data: Array<Int>): Int {

    var sum = 0
    //so when it is even number, square that number and add it to the sum
    for (item in data.indices) {
        if (item % 2 == 0) {
            sum += item * item
        }
    }
    return sum
}
